from langgraph.graph import StateGraph, END
from typing import TypedDict, List, Dict, Optional, Annotated
import operator
from datetime import datetime

class OpenPanelState(TypedDict):
    """State for Open Panel Type 2 execution"""
    
    # Core identifiers
    panel_id: str
    tenant_id: str
    user_id: str
    
    # Panel configuration
    query: str
    panel_type: str  # "open"
    agents: List[Dict]
    configuration: Dict
    
    # Execution state
    phase: str  # "opening" | "dialogue" | "clustering" | "synthesis"
    round_number: int
    start_time: datetime
    
    # Discussion data
    opening_statements: Annotated[List[Dict], operator.add]
    dialogue_turns: Annotated[List[Dict], operator.add]
    idea_units: Annotated[List[Dict], operator.add]
    
    # Analysis
    themes: List[Dict]
    clusters: List[Dict]
    convergence_points: List[Dict]
    divergence_points: List[Dict]
    
    # Consensus tracking
    consensus_level: float
    expert_alignments: Dict[str, float]
    
    # Outputs
    final_synthesis: Optional[str]
    innovation_map: Optional[Dict]
    deliverables: List[Dict]
    
    # Metadata
    total_duration: Optional[float]
    token_usage: int
    streaming_active: bool


class OpenPanelOrchestrator:
    """
    Orchestrator for Ask Panel Type 2: Open Panel
    Parallel collaborative exploration with dynamic turn-taking
    """
    
    def __init__(self):
        self.workflow = self._create_workflow()
    
    def _create_workflow(self) -> StateGraph:
        """Create LangGraph workflow for Open Panel"""
        
        workflow = StateGraph(OpenPanelState)
        
        # Define nodes
        workflow.add_node("initialize", self.initialize_panel)
        workflow.add_node("opening_round", self.execute_opening_round)
        workflow.add_node("free_dialogue", self.execute_free_dialogue)
        workflow.add_node("identify_themes", self.identify_themes)
        workflow.add_node("final_perspectives", self.gather_final_perspectives)
        workflow.add_node("synthesize", self.synthesize_recommendations)
        workflow.add_node("generate_deliverables", self.generate_deliverables)
        
        # Define edges
        workflow.set_entry_point("initialize")
        workflow.add_edge("initialize", "opening_round")
        workflow.add_edge("opening_round", "free_dialogue")
        workflow.add_edge("free_dialogue", "identify_themes")
        workflow.add_edge("identify_themes", "final_perspectives")
        workflow.add_edge("final_perspectives", "synthesize")
        workflow.add_edge("synthesize", "generate_deliverables")
        workflow.add_edge("generate_deliverables", END)
        
        return workflow.compile()
    
    async def initialize_panel(self, state: OpenPanelState) -> OpenPanelState:
        """Initialize panel with configuration and context"""
        
        logger.info(f"Initializing Open Panel {state['panel_id']}")
        
        # Load expert agents
        agents = await self.load_expert_agents(
            agent_ids=state['configuration']['agent_ids'],
            tenant_id=state['tenant_id']
        )
        
        # Load context documents if provided
        context = await self.load_context_documents(
            document_ids=state['configuration'].get('document_ids', []),
            tenant_id=state['tenant_id']
        )
        
        # Initialize streaming
        await self.initialize_sse_stream(
            panel_id=state['panel_id'],
            user_id=state['user_id']
        )
        
        state['agents'] = agents
        state['phase'] = 'opening'
        state['round_number'] = 1
        state['start_time'] = datetime.now()
        state['streaming_active'] = True
        
        # Stream initialization event
        await self.stream_event({
            'event': 'panel_initialized',
            'data': {
                'panel_id': state['panel_id'],
                'experts': [a['name'] for a in agents],
                'estimated_duration': '5-10 minutes'
            }
        })
        
        return state
    
    async def execute_opening_round(self, state: OpenPanelState) -> OpenPanelState:
        """Execute sequential opening statements from each expert"""
        
        logger.info(f"Executing opening round for panel {state['panel_id']}")
        
        state['phase'] = 'opening'
        opening_statements = []
        
        # Sequential opening statements (10-15 seconds each)
        for i, agent in enumerate(state['agents']):
            
            # Generate opening statement
            statement = await self.generate_expert_response(
                agent=agent,
                query=state['query'],
                context=None,  # No prior context in opening
                mode='opening_statement',
                max_tokens=150  # ~15 seconds when read
            )
            
            opening_statements.append({
                'agent_id': agent['id'],
                'agent_name': agent['name'],
                'content': statement,
                'timestamp': datetime.now().isoformat(),
                'sequence': i + 1
            })
            
            # Extract idea units from statement
            ideas = await self.extract_idea_units(statement)
            state['idea_units'].extend(ideas)
            
            # Stream to client
            await self.stream_event({
                'event': 'expert_speaking',
                'data': {
                    'agent_name': agent['name'],
                    'content': statement,
                    'phase': 'opening',
                    'sequence': i + 1,
                    'total_experts': len(state['agents'])
                }
            })
            
            # Brief pause between speakers (simulate natural flow)
            await asyncio.sleep(0.5)
        
        state['opening_statements'] = opening_statements
        
        # Stream phase completion
        await self.stream_event({
            'event': 'phase_complete',
            'data': {
                'phase': 'opening',
                'duration': '90 seconds',
                'next_phase': 'free_dialogue'
            }
        })
        
        return state
    
    async def execute_free_dialogue(self, state: OpenPanelState) -> OpenPanelState:
        """Execute dynamic free-form dialogue with idea building"""
        
        logger.info(f"Executing free dialogue for panel {state['panel_id']}")
        
        state['phase'] = 'dialogue'
        dialogue_turns = []
        
        # Configuration for dialogue phase
        max_turns = state['configuration'].get('max_dialogue_turns', 15)
        turn_duration_target = 180  # 3 minutes total
        
        # Track conversation state for intelligent turn-taking
        conversation_context = {
            'topics_discussed': [],
            'experts_spoken': {agent['id']: 0 for agent in state['agents']},
            'building_on_ideas': []
        }
        
        for turn in range(max_turns):
            
            # Select next speaker based on:
            # 1. Who has spoken least
            # 2. Who has relevant expertise for emerging topics
            # 3. Random chance for spontaneity
            next_speaker = await self.select_next_speaker(
                state=state,
                conversation_context=conversation_context,
                turn_number=turn
            )
            
            # Determine if this turn is:
            # - Building on previous idea (reference specific statement)
            # - Introducing new perspective
            # - Connecting multiple ideas
            turn_type = await self.determine_turn_type(
                speaker=next_speaker,
                conversation_context=conversation_context,
                recent_turns=dialogue_turns[-3:] if dialogue_turns else []
            )
            
            # Generate expert response with awareness of full conversation
            response = await self.generate_expert_response(
                agent=next_speaker,
                query=state['query'],
                context=self._build_dialogue_context(dialogue_turns, state['opening_statements']),
                mode='dialogue',
                turn_type=turn_type,
                max_tokens=200
            )
            
            turn_data = {
                'agent_id': next_speaker['id'],
                'agent_name': next_speaker['name'],
                'content': response,
                'turn_type': turn_type,  # 'building', 'new', 'connecting'
                'timestamp': datetime.now().isoformat(),
                'turn_number': turn + 1
            }
            
            dialogue_turns.append(turn_data)
            conversation_context['experts_spoken'][next_speaker['id']] += 1
            
            # Extract idea units
            ideas = await self.extract_idea_units(response)
            state['idea_units'].extend(ideas)
            
            # Identify if this creates new connections
            connections = await self.identify_idea_connections(
                current_response=response,
                previous_ideas=state['idea_units']
            )
            
            if connections:
                conversation_context['building_on_ideas'].extend(connections)
            
            # Stream to client
            await self.stream_event({
                'event': 'expert_speaking',
                'data': {
                    'agent_name': next_speaker['name'],
                    'content': response,
                    'phase': 'dialogue',
                    'turn_type': turn_type,
                    'turn_number': turn + 1,
                    'building_on': connections
                }
            })
            
            # Check if natural stopping point reached
            if turn >= 8:  # Minimum 8 turns
                should_continue = await self.assess_dialogue_continuation(
                    state=state,
                    dialogue_turns=dialogue_turns,
                    target_duration=turn_duration_target
                )
                
                if not should_continue:
                    logger.info(f"Dialogue phase naturally concluding at turn {turn + 1}")
                    break
            
            await asyncio.sleep(0.3)  # Brief pause between turns
        
        state['dialogue_turns'] = dialogue_turns
        
        # Stream phase completion
        await self.stream_event({
            'event': 'phase_complete',
            'data': {
                'phase': 'dialogue',
                'turns': len(dialogue_turns),
                'duration': '~3 minutes',
                'next_phase': 'theme_identification'
            }
        })
        
        return state
    
    async def identify_themes(self, state: OpenPanelState) -> OpenPanelState:
        """AI moderator identifies patterns, clusters, and themes"""
        
        logger.info(f"Identifying themes for panel {state['panel_id']}")
        
        state['phase'] = 'clustering'
        
        # Analyze all idea units to identify clusters
        clusters = await self.cluster_ideas(
            idea_units=state['idea_units'],
            opening_statements=state['opening_statements'],
            dialogue_turns=state['dialogue_turns']
        )
        
        # Identify convergence points (high agreement)
        convergence = await self.identify_convergence_points(
            clusters=clusters,
            expert_statements=state['opening_statements'] + state['dialogue_turns']
        )
        
        # Identify divergence points (creative tension)
        divergence = await self.identify_divergence_points(
            clusters=clusters,
            expert_statements=state['opening_statements'] + state['dialogue_turns']
        )
        
        state['clusters'] = clusters
        state['convergence_points'] = convergence
        state['divergence_points'] = divergence
        
        # Stream theme analysis
        await self.stream_event({
            'event': 'theme_analysis',
            'data': {
                'clusters': [
                    {
                        'theme': c['theme'],
                        'confidence': c['confidence'],
                        'key_components': c['components'][:3]  # Top 3
                    }
                    for c in clusters
                ],
                'convergence_areas': len(convergence),
                'divergence_areas': len(divergence)
            }
        })
        
        return state

    async def select_next_speaker(
        self, 
        state: OpenPanelState,
        conversation_context: Dict,
        turn_number: int
    ) -> Dict:
        """Intelligently select next speaker"""
        
        # Balance factors:
        # 1. Least spoken (fairness)
        # 2. Relevant expertise to emerging topics
        # 3. Random chance (spontaneity)
        
        agents = state['agents']
        speak_counts = conversation_context['experts_spoken']
        
        # Calculate scores for each agent
        scores = []
        for agent in agents:
            
            # Fairness score (inverse of how often they've spoken)
            min_speaks = min(speak_counts.values())
            agent_speaks = speak_counts[agent['id']]
            fairness_score = 1 / (agent_speaks - min_speaks + 1)
            
            # Relevance score (based on recent topics)
            relevance_score = await self.calculate_relevance_score(
                agent=agent,
                recent_topics=conversation_context['topics_discussed'][-3:]
            )
            
            # Random element
            random_score = random.random()
            
            # Weighted combination
            total_score = (
                fairness_score * 0.5 +
                relevance_score * 0.3 +
                random_score * 0.2
            )
            
            scores.append((agent, total_score))
        
        # Select agent with highest score
        scores.sort(key=lambda x: x[1], reverse=True)
        selected_agent = scores[0][0]
        
        return selected_agent
    
    async def determine_turn_type(
        self,
        speaker: Dict,
        conversation_context: Dict,
        recent_turns: List[Dict]
    ) -> str:
        """Determine what type of turn this should be"""
        
        # Analyze recent conversation flow
        if not recent_turns:
            return 'new'  # First turn in dialogue
        
        # Use LLM to classify turn type
        turn_type = await self.classify_turn_type(
            speaker=speaker,
            recent_turns=recent_turns,
            conversation_topics=conversation_context['topics_discussed']
        )
        
        return turn_type  # 'building', 'new', 'connecting'
    
    async def cluster_ideas(
        self,
        idea_units: List[Dict],
        opening_statements: List[Dict],
        dialogue_turns: List[Dict]
    ) -> List[Dict]:
        """Cluster ideas into themes using semantic similarity"""
        
        # Use embedding model to cluster related ideas
        embeddings = await self.get_idea_embeddings(idea_units)
        
        # Clustering algorithm (e.g., DBSCAN, K-means)
        clusters = await self.perform_clustering(
            embeddings=embeddings,
            idea_units=idea_units,
            min_cluster_size=3
        )
        
        # Generate theme names and descriptions for each cluster
        for cluster in clusters:
            theme_analysis = await self.analyze_cluster_theme(
                ideas=cluster['ideas'],
                supporting_statements=opening_statements + dialogue_turns
            )
            
            cluster['theme'] = theme_analysis['theme_name']
            cluster['description'] = theme_analysis['description']
            cluster['confidence'] = theme_analysis['confidence']
            cluster['key_components'] = theme_analysis['components']
            cluster['contributors'] = theme_analysis['contributing_experts']
        
        return clusters

## ğŸ“ LESSONS FROM REAL DEPLOYMENTS

### What Works Well

**1. Diverse Expert Mix**
- Technical + Clinical + Business + Design perspectives
- Cross-domain experts spark unexpected connections
- 6-8 experts is sweet spot (not too few, not too many)

**2. Minimal Moderation**
- AI facilitates but doesn't control
- Let conversation flow naturally
- Intervene only for theme identification

**3. Visual Innovation Mapping**
- Users love seeing idea connections visually
- Helps identify strongest clusters
- Makes abstract concepts concrete

**4. Quick Turnaround**
- 5-10 minutes feels "just right"
- Fast enough to maintain energy
- Long enough for depth to emerge

### Common Pitfalls to Avoid

**1. Over-Structuring**
- âŒ Don't force rigid turn-taking in dialogue phase
- âœ… Do allow natural conversation flow

**2. Premature Consensus**
- âŒ Don't push for agreement too early
- âœ… Do preserve creative tension and divergence

**3. Single-Domain Experts**
- âŒ Don't select all clinical or all technical experts
- âœ… Do intentionally create diverse perspective mix

**4. Vague Queries**
- âŒ "How can we improve our product?"
- âœ… "Generate innovative features for our mental health app that increase teen engagement to 60% 90-day retention"
