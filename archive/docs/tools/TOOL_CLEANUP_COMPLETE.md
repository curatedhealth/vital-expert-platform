```sql

FROM dh_tool
WHERE category = 'Healthcare/Clinical NLP'
  AND lifecycle_stage = 'production';
```

### **Search by Tags:**
```sql
SELECT name, category, tags
FROM dh_tool
WHERE 'hipaa' = ANY(tags)
   OR 'phi' = ANY(tags);
```

### **Parent Category Statistics:**
```sql
SELECT 
  category_parent,
  COUNT(*) as tool_count,
  COUNT(DISTINCT category) as subcategories
FROM dh_tool
GROUP BY category_parent
ORDER BY tool_count DESC;
```
